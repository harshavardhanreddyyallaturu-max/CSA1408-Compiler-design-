%{
#include <stdio.h>

int vowels = 0, consonants = 0;
%}

%%
[aAeEiIoOuU]      { vowels++; }

[a-zA-Z]          { consonants++; }

\n                { return 0; }

.                 ;
%%

int yywrap()
{
    return 1;
}

int main()
{
    printf("Enter a string:\n");
    yylex();

    printf("\nNumber of vowels     = %d\n", vowels);
    printf("Number of consonants = %d\n", consonants);

    return 0;
}
